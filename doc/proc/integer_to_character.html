      function integer_to_character(i) result(res)
         !! transfers integer to a character
         !---------------------------------------------------------------------!
         integer, intent(in) :: i
            !! input integer
         character(len=32) :: res
            !! output character
         !---------------------------------------------------------------------!
         write (res, '(i0)') i
         res = adjustl(res)
         !---------------------------------------------------------------------!
      end function integer_to_character
